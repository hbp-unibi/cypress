251 template <typename T>
260 void from_json(const Json &j, Matrix<T> &mat)
261 {
262  size_t rows = j.size();
263  if (rows == 0) {
264  return;
265  }
266  size_t cols = j[0].size();
267  mat = Matrix<T>(rows, cols);
268  for (size_t i = 0; i < rows; i++) {
269  for (size_t k = 0; k < cols; k++) {
270  mat(i, k) = T(j[i][k]);
271  }
272  }
273 }
